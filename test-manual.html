案件管理テスト用ページ

このページをブラウザで開いて、以下のJavaScriptをコンソールで実行してください：

// 1. マスターページを新しいタブで開く
const testWin = window.open('https://3000-ip60pndnnf5e5cmovjufh-6532622b.e2b.dev/masters', '_blank');

// 2. 3秒後に案件タブをクリック
setTimeout(() => {
  if (testWin.MasterManagement && testWin.MasterManagement.switchTab) {
    console.log('案件タブをクリック中...');
    testWin.MasterManagement.switchTab('projects');
    
    // 3. さらに2秒後に新規追加ボタンをクリック
    setTimeout(() => {
      const addBtn = testWin.document.getElementById('addProjectBtn');
      if (addBtn) {
        console.log('新規追加ボタンをクリック中...');
        addBtn.click();
      } else {
        console.error('新規追加ボタンが見つかりません');
      }
    }, 2000);
    
  } else {
    console.error('MasterManagement が見つかりません');
  }
}, 3000);
